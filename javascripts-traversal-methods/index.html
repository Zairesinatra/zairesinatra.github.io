for、for...in、for...of、forEach

Ziyi Xie

Sep 25, 2019 • 2 min read

分类描述

for (let i = 0; i < arr.length; i++) 和 for (let i in arr) 中的形参访问数组下标，而不是实际数组元素值。

const zsarr = [5,4,3,2,1,0]
for (let i = 0; i < zsarr.length; i++) {
    console.log(zsarr[i]); // 5 4 3 2 1 0
    console.log(i); // 0 1 2 3 4 5
}

for (let i in zsarr) {
    console.log(arr[i]); // 5 4 3 2 1 0
    console.log(i); // 0 1 2 3 4 5
}

// 注意 for...in 常用于遍历对象, 且遍历的每一项为 key 键
var obj={"name":"xzy","age":23,"sex":"male"}; // 字面量定义 object 对象
var keys=[]; // 数组用来接受 key    
var values=[]; // 数组用来接受 value
for(var key in obj){
  keys.push(key);    
  values.push(obj[key]); // 取得 value
}    
console.log("keys is ："+keys+" values is ："+values); // keys is ：name,age,sex values is ：xzy,23,male

for (const v of arr)、arr.forEach((v) => { func }) 访问数组的元素值。

const zsarr = [5,4,3,2,1,0]
for (const v of zsarr) {
    console.log(v); // 5 4 3 2 1 0
}
// 第1形参表示数组元素; 第2形参 i 访问数组的下标
zsarr.forEach((v, i) => console.log(v, i)); // v: 5 4 3 2 1 0 ; i: 0 1 2 3 4 5

使用特性

for...in 不会忽略非数字属性，其他历遍都会忽略添加属性。简而言之，使用 for(let i in arr) 会遍历数组所有的可枚举属性，包括原型上属性与方法。

// 数组是可以添加属性的 Object
Array.prototype.myfun=function(){
  console.log('使用for-in会遍历数组所有的可枚举属性,包括原型(原型上方法与属性)');
}
var zsarr = ["x", "z", "y"]
zsarr.test = "Less used for in";
for(var i in zsarr){
  console.log(zsarr[i]); // x z y Less used for in [Function (anonymous)]
}

for (let i in arr) 与 arr.forEach((v, i) => { func }) 会跳过空元素；而对于 undefined 则上述历遍方式都不会跳过。

const zsarr = [1, ,3];
const zyarr = [1, 2, 3];
zyarr[5] = 5;
for (let i in zsarr) {
  console.log(i); // 0 2
}
for (let i in zyarr) {
  console.log(i); // 0 1 2 5
}

forEach 遍历时的 this ：

// 非严格模式
const zsarr = [1];
zsarr.forEach(function() {
    console.log(this === globalThis); // true
    console.log(this);
    /*

// 严格模式
"use strict";
const zsarr = [1];
zsarr.forEach(function() {
    console.log(this === undefined); // true
});
