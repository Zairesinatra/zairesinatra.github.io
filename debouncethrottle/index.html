Debounce&Throttle

星星发亮是为了让每一个人有一天都能找到属于自己的星星 ——《小王子》

Ziyi Xie

Sep 20, 2020 • 4 min read

防抖策略(debounce)

意义： 保证频繁触发事件时，事件回调仅被触发一次。减少请求次数、节约请求资源。

定义： 当事件被触发后，延迟n秒再执行回调，若在这n秒内事件又被触发则重新计时。——类似游戏英雄联盟的回城操作。

应用： 输入框防抖-淘宝搜索框中连续输入字符，保证仅在输入完成后才进行查询的请求。

不采用防抖问题复现：

// 如果不加防抖函数,chrome控制台每当你按下键盘则输出
<input type="text" class="ipt">

const ipt = document.querySelector('.ipt')
ipt.onkeyup = function(){
    console.log(ipt.value)
}

采用防抖解决：

let timer = null
ipt.onkeyup = function(){
    if(timer){
        clearTimeout(timer)
    }
    timer = setTimeout(()=>{
        console.log(ipt.value)
    },2000)
}

由于以上代码可以抽离出来封装自己的业务逻辑。避免每遇见防抖又得手写一遍。封装成debounce：

function debounce(fn,delay=2000) { // 传入自定义函数与延迟;延迟不传入设默认值
    let timer; // const必须初始化,这里用let
    return ()=>{ // 用外函数的timer形成闭包
        if(timer){clearTimeout(timer)} // 有timer
        timer = setTimeout(() => { // 没timer
            fn();
        }, delay);
    }
}
ipt.onkeyup = debounce(()=>{console.log(ipt.value)},1500)

节流策略(throttle)

意义： 解决因频繁触发可能造成的卡顿现象。

定义： 连续发生的事情在n秒内只执行一次回调函数。——类似射击游戏进行连续点击，但枪械存在固定射速。

应用： 拖拽效果的节流、云服务器更新内存使用比率、移动端后台刷新定位频率。

不采用节流问题复现：

.zs{
    width: 50px;
    height: 50px;
    background-color: #424242;
    border-radius: 50%;
    color:red;
    font-size: 5px;
    text-align: center;
}

<div class="zs" draggable="true">zy</div>
<script>
    const zs = document.querySelector('.zs')
    zs.ondrag = function(e){
        console.log(e.offsetX,e.offsetY)
    }
</script>

采用节流解决：

const zs = document.querySelector('.zs')
let timer = null;
zs.ondrag = function(e){
    if(timer){return;}
    timer = setTimeout(()=>{
        console.log(e.offsetX,e.offsetY)
        timer = null
    },2000)
}

同上，为复用封装为throttle函数。

const zs = document.querySelector('.zs')
function throttle(fn,delay=2000){
    let timer
    return function(){
        if(timer) {return;}
        // console.log(arguments)
        fn.apply(this,arguments)
        timer = setTimeout(()=>{fn;timer=null},delay)
    }
}
zs.ondrag = throttle((zy)=>{console.log(zy.offsetX,zy.offsetY)},1500)

防抖与节流的指向

<div id="content" style="height:150px;line-height:150px;text-align:center; color: #fff;background-color:#ccc;font-size:80px;"></div>

// 防抖
let num = 1;
let content = document.getElementById('content');
function count(e) {
    console.log(this); // <div id="content" style="height:150px;line-height:150px;text-align:center; color: #fff;background-color:#ccc;font-size:80px;">2</div>
    console.log(e); // MouseEvent {isTrusted: true, screenX: 691, screenY: 217, clientX: 691, clientY: 81, …}
    content.innerHTML = num++;
};
function debounce (func, delay) {
    let timer = null;
    return function() {
        // this也是指向的DOM元素,事件参数也是绑定的onmousemove
        if(timer) {
            clearTimeout(timer)
        }
        timer = setTimeout(() => {
            func.apply(this, arguments)
            // func() // Window {window: Window, self: Window, document: document, name: '', location: Location, …} undefined
            timer = null
        }, delay)
    }
}
// content.onmousemove = count;
content.onmousemove = debounce(count, 1500);

// 节流
let num = 1;
let content = document.getElementById('content');
function count(e) {
    console.log(this); // <div id="content" style="height:150px;line-height:150px;text-align:center; color: #fff;background-color:#ccc;font-size:80px;">2</div>
    console.log(e); // MouseEvent {isTrusted: true, screenX: 691, screenY: 217, clientX: 691, clientY: 81, …}
    content.innerHTML = num++;
};
function throttle (func, delay) {
    let timer = null;
    return function() {
        // this也是指向的DOM元素,事件参数也是绑定的onmousemove
        if(timer) {
            return
        }
        timer = setTimeout(() => {
            func.apply(this, arguments)
            // func() // Window {window: Window, self: Window, document: document, name: '', location: Location, …} undefined
            timer = null
        }, delay)
    }
}
// content.onmousemove = count;
content.onmousemove = throttle(count, 1500);
