手写 Object.create

new 和 Object.create 都用于创建对象，但前者执行构造函数，后者不执行，且使用现有的对象来提供新创建的对象的 __proto__。

Male.prototype.constructor=Male;
Male.prototype.say=function(){
	alert("Male func say hello world");

var m = new Male();
console.log(m instanceof Male); // true
console.log(m instanceof Human); // true

// 思路 => 将传入的对象作为原型
function create(obj) {

手写 new

在调用 new 的过程里，首先会在内存中创建一个新对象，再将新对象的原型设置为函数的 prototype 对象，以继承父类原型上的方法，然后让 this 指向这个新对象，执行构造函数里的代码，给这个新对象添加属性和方法，最后判断函数的返回值类型，如果是值类型就返回创建的对象；如果是引用类型，就返回这个引用类型对象。

function objectFactory() {
  let newObject = null;
  let constructor = Array.prototype.shift.call(arguments);
  let result = null;
  // 判断参数是否是一个函数
  if (typeof constructor !== "function") {
    console.error("type error");
    return;
  }
  // 新建一个空对象，对象的原型为构造函数的 prototype 对象
  newObject = Object.create(constructor.prototype);
  // 将 this 指向新建对象，并执行函数
  result = constructor.apply(newObject, arguments);
  // 判断返回对象
  let flag = result && (typeof result === "object" || typeof result === "function");
  // 判断返回结果
  return flag ? result : newObject;
}
// 使用方法
objectFactory(构造函数, 初始化参数);

手写 instanceof

instanceof 运算符用于检测构造函数的 prototype 属性是否出现在某个实例对象的原型链上。

// 获取类型的原型 => 获得构造函数与实例的原型 => 循环判断实例的原型是否等于构造函数的原型 => 不等于往上找,直到对象原型为 null,因为原型链最终为 null.
function myInstanceof(arg1, arg2) {
    // 获取实例的原型以及构造函数的原型
    // Object.getPrototypeOf() 方法返回指定对象的原型
    let pro = Object.getPrototypeOf(arg1),prot = arg2.prototype;
    // 判断构造函数的 prototype 对象是否在对象的原型链上
    while (true) {
        // pro非null
        if (!pro) return false;
        if (pro === prot) return true;
        pro = Object.getPrototypeOf(pro);
    }
}

  this.make = make;

  this.year = year;
}
const auto = new Car('Honda', 'Accord', 1998);
console.log(myInstanceof(auto,Car)); // true

防抖 debounce

当事件被触发后，延迟n秒再执行回调，若在这n秒内事件又被触发则重新计时。——类似游戏英雄联盟的回城操作。保证频繁触发事件时，事件回调仅被触发一次。减少请求次数、节约请求资源。常应用于输入框防抖，连续输入字符，保证仅在输入完成后才进行查询的请求。

// 如果不加防抖函数,chrome控制台每当你按下键盘则输出
<input type="text" class="ipt">
...
const ipt = document.querySelector('.ipt')
ipt.onkeyup = function(){
    console.log(ipt.value)
}

// 采用防抖解决
let timer = null
ipt.onkeyup = function(){
    if(timer){
        clearTimeout(timer)
    }
    timer = setTimeout(()=>{
        console.log(ipt.value)
    },2000)
}

// 抽离封装业务逻辑
function debounce(fn,delay=2000) { // 传入自定义函数与延迟;延迟不传入设默认值
    let timer; // const必须初始化,这里用let
    return ()=>{ // 用外函数的timer形成闭包
        if(timer){clearTimeout(timer)} // 有timer
        timer = setTimeout(() => { // 没timer
            fn();
        }, delay);
    }
}
ipt.onkeyup = debounce(()=>{console.log(ipt.value)},1500)

节流 throttle

节流即连续发生的事情在n秒内只执行一次回调函数。类似射击游戏进行连续点击，但枪械存在固定射速。解决因频繁触发可能造成的卡顿现象。常应用于拖拽效果的节流、云服务器更新内存使用比率、移动端后台刷新定位频率。

.zs{
    width: 50px;
    height: 50px;
    background-color: #424242;
    border-radius: 50%;
    color:red;
    font-size: 5px;
    text-align: center;
}

// 不采用节流
<div class="zs" draggable="true">zy</div>
...
const zs = document.querySelector('.zs')
zs.ondrag = function(e){
    console.log(e.offsetX,e.offsetY)
}

// 采用节流
const zs = document.querySelector('.zs')
let timer = null;
zs.ondrag = function(e){
    if(timer){return;}
    timer = setTimeout(()=>{
        console.log(e.offsetX,e.offsetY)
        timer = null
    },2000)
}

// 为复用封装为throttle函数
const zs = document.querySelector('.zs')
function throttle(fn,delay=2000){
    let timer
    return function(){
        if(timer) {return;}
        // console.log(arguments)
        fn.apply(this,arguments)
        timer = setTimeout(()=>{fn;timer=null},delay)
    }
}
zs.ondrag = throttle((zy)=>{console.log(zy.offsetX,zy.offsetY)},1500)

栈与堆

栈是内存中一片连续用于存储局部变量和函数参数的线性结构。严格先进后出，且入栈和出栈的操作仅是栈指针在内存地址中的上下移动。栈在函数调用时创建，调用结束则消失。

堆是内存中一大块独立存储空间，在栈中存储不了的数据（对象），就会被存储在堆中，而栈中仅存对该数据的引用（也就是该块数据的首地址）。为避免内存溢出，在 C 中堆内存是需要手动清理的，而在 JAVA 之流存在 Garbage Collection 机制协助清理堆中已不被使用的数据。

function Closure () {
    let num = 0;
    return function () {
        num++;
        return num;
    }
}
let Closuree = Closure();
Closuree(); // 1
Closuree(); // 2

观察这段闭包代码是否让你顿时怀疑人生，因为上述栈的描述里，在调用完函数后，栈会销毁，那么就不应该存在后续的 num 仍然会发生改变。实际上在本例中的 num 变量并没有存储在栈中，而是在堆内存里，用一个特殊的对象 [[Scopes]] 保存。

在 JavaScript 中，变量分为三种类型：局部变量、被捕获变量、全局变量。其中 被捕获变量 较为少见——在函数中声明，却在函数返回后仍有未执行作用域（函数或是类）使用到该变量，那么该变量就是被捕获变量（可以简单理解为闭包中返回带有的参数变量）。

总结：只有局部变量的基本数据类型的值是存放在栈中的，其他（即其他的基本数据类型和所有的引用类型）都是地址存放在栈中，值存放在堆中。

深、浅拷贝

从上述可以得知，深、浅拷贝是对于引用类型来说的，引用类型的值实际存储在堆 Heap 内存中，在栈 Stack 中只存储堆内地址的引用，而这个地址的引用指向堆内存中的值。

浅拷贝仅拷贝一层，对更深层次对象级别只拷贝引用，那么在操作数据时难免出现对象互相影响的情况。深拷贝需要拷贝每一层的所有的属性，且地址也与原来的不同，那么改变属性时也就不会影响原来的数据。

function deepCopy(newobj,oldobj){
    for(var i in oldobj){
        // 获取属性值
        var hhh = oldobj[i];
        // 判断数据属于那种数据类型
        // 属于数组
        if(hhh instanceof Array){
            // 新对象属性是数组
            newobj[i]=[];
            // deepCopy(newobj[i],oldobj[i])
            deepCopy(newobj[i],hhh)
        }
        // 属于对象
        else if(hhh instanceof Object){
            newobj[i]={};
            // 把值给属性
            deepCopy(newobj[i],oldobj[i])
        }
        // 属于简单数据类型
        else {
            newobj[i] = oldobj[i]
        }
    }
}
deepCopy(g,obj);

// 斐波那契数列: 0 1 1 2 3 5 8 13 ...(0不是第一项,而是第0项)
function fibonacci(n) {
    if(n==0 || n == 1)
        return n;
    return fibonacci(n-1) + fibonacci(n-2);
}
console.log(fibonacci(3)); // 2
console.log(fibonacci(5)); // 5
console.log(fibonacci(7)); // 13

实现数组扁平化

数组扁平化就是将多维数组转化为一维数组。

// 方法一 => toString + split => 数组-字符串-数组
let arr = [1, [2, 3, [4, 5]], 6]
function flatten(arr) {
    // 用split形成的数组的每个元素仍然是字符串,需要将其转化为数字.
    return arr.toString().split(',').map( item => Number(item))
}
console.log(flatten(arr)) // [1, 2, 3, 4, 5]
// 方法二 => join + split => 数组-字符串-数组
function flatten(arr) {
    return arr.join(',').split(',').map( item => parseInt(item) )
}
// 方法三 => reduce 方法根据回调函数对数组的每个元素进行操作
let arr = [1, [2, 3, [4, 5]], 6]
const newArr = function(arr){
    return arr.reduce((pre,cur)=>pre.concat(Array.isArray(cur)?newArr(cur):cur),[])
}
console.log(newArr(arr)); //[1, 2, 3, 4, 5, 6]
// 方法四 => 递归与循环
function flatten(arr) {
    var res = [];
    arr.map(item => {
        if(Array.isArray(item)) {
            res = res.concat(flatten(item));
        } else {
            res.push(item);
        }
    });
    return res;
}

对象扁平化

对象扁平化就是将复杂的多层的对象转化为一维对象。

let obj = { a: { d: 1 }, 
            d:  2, 
            e: [1, 2] 
          }
function flatten(data) {
    var result = {};
    function recurse(cur, prop) {
        // 如果输入进来的是不是对象，就将其放在数组中，返回
        if (Object(cur) !== cur) {
            result[prop] = cur;
        // 如果输入进来的是数组，长度不为0就递归数组，得出结果
        } else if (Array.isArray(cur)) {
            for (var i = 0, len = cur.length; i < len; i++)
                recurse(cur[i], prop + "[" + i + "]");
            if (len == 0)  result[prop] = [];
        } else {
            var isEmpty = true;
            for (var p in cur) {
                isEmpty = false;
                recurse(cur[p], prop ? prop + "." + p : p);
            }
            if (isEmpty && prop)
                result[prop] = {};
        }
    }
    recurse(data, "");
    return result;
};
console.log(flatten(obj))  // {a.d: 1, d: 2, e[0]: 1, e[1]: 2}
