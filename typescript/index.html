关于 TypeScript 那些事

Ziyi Xie

May 1, 2020 • 12 min read

JavaScript 是动态弱类型类型的语言，在快速上手同时也存在很多令开发者诟病的缺点。使用变量可以赋值任意类型，但在不需要考虑类型的同时，若在其他地方大意改变了值的类型会导致产生让人费解的 BUGS（运算中对参数的数值变量重新赋值产生 NaN 的结果）。这种原因不会在写代码的时候报错，排查起来比较麻烦。进一步来说，在函数传参时没有类型的限制会导致在后续代码埋下安全隐患。

快速上手

TypeScript 可以在任何支持 JavaScript 的平台执行，但是 TS 不能被 JS 解析器直接执行，需要进行编译成最终的 JS 代码。

# 安装 TypeScript
npm install -g typescript
# 查看 TypeScript 版本
tsc -v
# 运行 TypeScript 编译器
tsc xxx.ts
# 防止出现编辑器提示 [函数重复实现]和[无法重新声明块范围变量]
tsc --init # 生成一个tsconfig文件即可

// 变量类型声明
let a: number; a = 10;
// 变量声明赋值同时进行则可以省略类型声明 let b: boolean = true;
let b = true;
console.log(a, b);

function greeter(person: string, age: number): string { // 定义参数类型与返回值类型
    return "Hello, " + person + '--' +age;
}
let person = "zairesinatra", age = 23;
console.log(greeter(person, 23)); // Hello, zairesinatra--23

TS 中的类型

// 可以直接使用字面量进行类型声明
let a: 10;
a = 10;
a = 11;❌
// 使用管道符连接多个类型 —— 联合类型
let b = "male" | "female";
let list: number[] | Array<number>; // 在元素类型后面接上[],表示由此类型元素组成的一个数组 | 使用数组泛型
list = [1, 2, 3];
// any表示任意类型,一个变量设置任意类型表示关闭了 TS 的类型检测
let d: any // 声明变量不指定类型也会隐式解析为 any 类型
d = '*'
// unknown 表示未知类型
let e: unknown; // any 可以赋值给任意变量; unknown 实际上是类型安全的 any
a = e;❌ // unknown 类型的变量不能直接赋值给其他变量,除非赋值前进行类型检查或者类型断言
a = e as number; // if( typeof e === 'number'){a = e} or a = <number>e
// void 表示函数无返回值,或者说返回空值
function voidfunc () {
    return 
}
// never 表示永远不会返回结果
function neverFunc (): never {
    throw new Error('your func has no return value')
}
// object 表示一个js对象,由于不能指定具体对象类型使用较少
let obj: object;
obj = {}; obj = () => {}
// {} 常做指定对象中可以包含的属性 {属性名: 属性类型, ...}
let objj: {name: string, age?: number} // 属性名后加上 ? 表示属性可选
objj = {name: 'zszs', age: 23}
// 中括号表示任意属性
let objjj: {name: string, [propName: string]: any} // 可选任意属性名的任意属性
objjj = {name: 'zszs', age: 23, gender: 'male'}
// 设置函数结构的类型声明 (形参: 类型, ...) => 返回值类型
let func: (arg1: number, arg2: number) => number;
func = function(arg1, arg2) {return arg1 + arg2}
console.log(func(1, 2)); // 3
// TS 数组通常存储相同类型的值(类型后加[];Array<类型>)
let arr: string[] // 等价于 let arr: Array<string>
arr = ['z', 'y']
// 元组: 固定长度的数组(存储效率较好) —— Tuple
let tuple: [boolean, string]
tuple = [true, 'zyyy']
// enum 枚举
enum Gender {
    Male = 0,
    Female = 1
}
let enumtest: {name: string, gender: Gender};
enumtest = {
    name: 'Kakashi',
    gender: Gender.Male // 0
}
console.log(enumtest.gender === Gender.Male) // true
// & 表示同时满足连接的条件
let j: {name: string} & {age: number};
j = {name: 'kakashi', age: 24}
// 类型别名 —— 简化类型书写与使用
type myType = 1 | 2 | 3 | 4 | 5;
let k: myType;
k = 1

编译选项

# 监视 TS 文件改变自动编译
tsc file.ts -w # 等同于 --watch
# 全局监视(根目录存在 tsconfig.json )
tsc -w

// tsconfig.json
// include 指定哪些 ts 文件需要进行编译
"include": [
    "./src/**/*" // **任意目录 *任意文件
],
// exclude 指定哪些文件不需要进行编译
"exclude": [
    "./src/exc/**/*" // 默认排除 ["node_modules", "bower_component", "jspm_packages"]
],
// extends 定义被继承的配置文件
"extend": "./config/base", // 当前配置文件会自动包含 config 目录下 base.json 中的所有配置信息
// 在指定的编译文件较少时,可以使用 files 指定需要编译的内容
"files": [
    "core.ts",
    "sys.ts",
    ...
],
// 在初始化tsc时会自动建立 compilerOptions 配置项
// compilerOptions 决定了编译器如何对 ts 文件进行编译
"compilerOptions": {
    "target": "es5", // 指定 ts 文件被编译为的 ES 版本
    "module": "commonjs", // 编译后的 js 需要使用的模块化规范
    // "lib": [], // 指定项目中需要用到的库,一般情况不动
    // outDir用来指定编译后文件所在目录
    "outDir": "./dist",
    // outFile 可以将代码合并为一个文件(模块化必须是 AMD 或者 system)
    // "outFile": "./" // 初始化默认注释
    // allowJs 是否对 JS 文件进行编译, 默认为 true ,但是在初始化时被注释
    "allowJs": true,
    // 检查 JS 代码是否符合规范
    "checkJs": true, // 通常与 allowJs 一套
    // removeComments 是否移除注释 —— 编译时是否携带注释
    "removeComments": true,
    // noEmit 不生成编译后的文件
    "noEmit": false, // 不想用编译功能,只需要检查语法
    // noEmitOnError 当有错误时不生成编译后的文件
    "noEmitOnError": false,
    // 所有严格检查的总开关 -- 设置为 true 则所有严格模式全部打开
    "strict": true,
    // alwaysStrict 编译后文件是否使用严格模式
    "alwaysStrict": false, // js 使用导入导出就会默认开启严格模式所以不会显示 "use strict"
    // 不允许隐式的 any 类型
    "noImplicitAny": true,
    // 不允许不明确类型的 this
    "noImplicitThis": true,
    // 严格检查空值
    "strictNullChecks": false
    }
}

// webpack中的所有的配置信息都应该写在module.exports中
module.exports = {
    // 指定入口文件
    entry: "./src/index.ts",
    // 指定打包文件所在目录
    output: {
        // 指定打包文件的目录
        path: path.resolve(__dirname, 'dist'),
        // 打包后文件的文件
        filename: "bundle.js",
        // 让 webpack 不使用箭头 —— 默认打包后是一个立即执行的箭头函数(在 IE11 中无法执行
        environment: {
            arrowFunction: false // 在webpack打包时,最外层不再是箭头函数
        }
    },
    // 指定webpack打包时要使用模块
    module: {
        // 指定要加载的规则
        rules: [
            { // Webpack在加载时是从后向前加载
                test: /\.ts$/, // test指定的是规则生效的文件
                use: [
                    {
                        loader: "babel-loader", // 指定加载器loader —— 与 webpack 结合的工具
                        options: {
                            presets: [ // 设置预定义的环境
                                [
                                    "@babel/preset-env", // 指定环境的插件
                                    {
                                        targets: { // 目标兼容的目标浏览器
                                            "chrome": "93",
                                            "ie": "11"
                                        },
                                        "corejs": "3", // 指定 corejs 的版本去兼容老版本浏览器
                                        "useBuiltIns": "usage" // 使用 corejs 按需加载
                                    }
                                ]
                            ]
                        }
                    },
                    'ts-loader', // ts 转换为 js
                ],
                exclude: /node-modules/ // 要排除的文件
            }
        ]
    },
    plugins: [
        new CleanWebpackPlugin(),
        new HTMLWebpackPlugin({ // html 文件的自定义配置
            // title: "这是一个自定义的title"
            template: "./src/index.html" // 根据此 html 模板生成页面
        }),
    ],
    resolve: { // 设置引用模块 —— 默认识别不出 .ts 结尾的模块
        extensions: ['.ts', '.js'] // 尝试按顺序解析这些后缀名
    }
}

面向对象 TS

/**
 * 直接定义的属性是实例属性,需要对象实例去访问
 * 使用 static 开头的属性是静态属性,可以直接通过类访问
 * readonly 开头属性为只读属性,无法修改
 */
class Person {
    readonly name: string = 'kakashi'; // 不可被修该 —— 只读属性
    static age: number = 24;
}
const per = new Person();
console.log(Person.age); // 24
console.log(per.name); // kakashi

// ts 会使不同文件中的相同变量出现冲突,使用立即执行函数规避
(function(){
    class Fruit {
        name: string;
        constructor(name: string) {
            this.name = name;
        }
        sayHello() {
            console.log('水果好吃')
        }
    }
    class Coconut extends Fruit {
        age: number;
        constructor(name: string, age: number) { // 创建属性初始化必须调用构造函数去创建
            super(name)
            // 由于名字相同的函数会发生重写,在子类写构造函数相当于把父类构造函数覆盖掉了,那么父类构造函数不会执行,会导致 name 赋值失效,也就是继承失效
            this.age = age
        }
        sayHello(){
            super.sayHello(); // super 表示当前类的父类
            console.log(`other methods: 我是一个${this.age}岁的${this.name}水果`); // other methods: 我是一个0.2岁的coco水果
        }
    }
    const coco = new Coconut('coco', 0.2)
    coco.sayHello(); // 水果好吃 —— 实际在调用父类的 sayHello
})()

(function(){
    /**
     * 以 abstract 开头的类是抽象类
     * 抽象类与其他类差别不大,仅不能创建对象
     * 抽象类是专门被其他类继承的类
     * 抽象类可以添加抽象方法
     */
    abstract class Animal {
        name: string;
        constructor(name: string) {
            this.name = name;
        }
        /**
         * 具体实现由子类实现
         * 抽象方法由 abstract 开头,没有方法体
         * 抽象方法只能定义在抽象类中,子类必须对抽象方法进行重写
         */
        abstract sayHello(): void; // 通用方法不能满足每个类的需求
    }
    class Cat extends Animal{
        sayHello() { // 必须重写抽象方法
            console.log(" cat class rewrite this method");
        }
    }
    // const cat = new Animal() // 无法创建抽象类的实例
})()

(function(){
    /**
     * 接口用来定义限制一个类的结构
     * 定义一个类中应该包含哪些属性与方法,同时接口可以作为类型声明去使用 => type myType = {name:string, age: number} !但是接口可以重复声明
     */
    interface myInterface{ // 类似于类型的方式
        /**
         * 接口中所有的属性都不能有实际的值,即接口只定义结构不考虑实际值
         */
        name: string;
        age: number;
        sayHi(): void;
    }
    const obj: myInterface = {
        name: 'zszy',
        age: 23,
        sayHi(){
            console.log('implements interface');
        }
    }
    /**
     * 定义类时,可以使类去实现一个接口
     * 实现接口即使类满足接口的要求
     */
    class MyClass implements myInterface{
        constructor(name: string, age:number) {
            this.name = name;
            this.age = age
        }
        sayHi(): void {
            throw new Error("Guess method is not implemented.");
        }
        name: string
        age: number
    }
})()

"use strict";
/**
 * 属性在对象中设置,属性可以任意的修改
 * 属性被任意的修改会导致数据安全性下降
 */
(function () {
    /**
     * ts 可以在属性前添加属性的修饰符
     * public 的属性可以在任意位置修改与访问(是默认的修饰)
     * private 为私有属性,只能在类内部进行修改
     *  -- 通过在类中添加私有属性可以被外部进行访问 —— 通过暴露方法间接得到属性
     *  -- 属性访问权回收,保证数据安全
     */
    var Person = /** @class */ (function () {
        function Person(name, age) {
            this._name = name;
            this._age = age;
        }
        /**
         * getter 与 setter 方法称为属性存取器
         * TS 中可以简化设置
         */
        Person.prototype.getName = function () {
            return this._name;
        };
        Person.prototype.setName = function (value) {
            this._name = value;
        };
        Person.prototype.getAge = function () {
            return this._age;
        };
        Person.prototype.setAge = function (value) {
            if (value >= 0) {
                this._age = value;
            }
        };
        return Person;
    }());
    var p = new Person('Sasuke', 18);
    /**
     * private 修饰属性被外部禁止修改
     * ts 独有,若不进行配置 "noEmitOnError": true 会导致还是会在解析的JS文件被修改
     */
    // p.age = -1; // Person { name: 'Sasuke', age: -1 }
    p.setName('kakashi');
    p.setAge(24);
    console.log(p); // Person { _name: 'kakashi', _age: 24 }
})();

由 TS 存取器简化后

(function(){
    /**
     * 属性容易修改错则在开发中使用
     */
    class Person {
        private _name: string;
        private _age: number;
        constructor(name: string, age: number) {
            this._name = name;
            this._age = age;
        }
        // TS 中设置的 getter setter 方法
        get name() {
            return this._name
        }
        set name(value: string) {
            this._name = value
        }
        get age() {
            return this._age
        }
        set age(value: number) {
            if(value >= 0){
                this._age = value;
            }
        }
    }
    const p = new Person('Sasuke', 18)
    console.log(p.name, p.age); // Sasuke 18
    p.name = 'Naruto'; p.age = 17;
    console.log(p.name, p.age); // Naruto 17
})()

/**
 * 需要少使用 any,又难以避免定义函数或者类遇见类型不明确的情况,故使用泛型 —— 根据实际调用情况决定类型
 */
function fn<T>(a: T): T{ // T只有在函数执行时候才确定
    return a
}
// 直接调用具有泛型的函数 —— 不指定泛型可以让 ts 自动对类型推断
fn(10); // 返回值类型也是 number
// 指定泛型
fn<string>('hello'); // 返回值类型也是 string

// 泛型可以指定多个
function func<T, K>(a: T, b: K){
    console.log(b);
    return a
}
func<number, string>(123, 'hello')

// T extends Inter 表示泛型必须是 Inter 实现类(子类)
interface Inter{
    length: number;
}
function fun<T extends Inter>(a: T): number{
    return a.length;
}
fun('123')
